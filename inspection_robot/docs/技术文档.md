#### 2.1.2 核心算法

##### 抱紧机构计算
```python
# 公式3-1 ~ 3-8
def calc_stroke_from_radius(R_values):
    # 接触条件：L2 * sin(theta) = R + WHEEL_R
    ratio = np.clip((R + cls.WHEEL_R) / cls.L2, -1.0, 1.0)
    theta = np.arcsin(ratio)
    # 推程：S = 2 * L1 * (cos(theta_min) - cos(theta))
    S = 2.0 * cls.L1 * (np.cos(theta_min) - np.cos(theta))
    return np.clip(S, 0, None)
```

##### 驱动电机扭矩
```python
# 公式3-25 ~ 3-32
def calc_torque_slope(weight, beta_deg, mu_roll=0.02):
    # 爬坡阻力 = 重力分量 + 滚动阻力
    F_grav = weight * g * np.sin(beta)
    F_fric = weight * g * np.cos(beta) * mu_roll
    F_total = (F_grav + F_fric) / n_motors
    # 电机扭矩：T_motor = T_out / (gear_ratio * eta)
    T_motor = F_total * WHEEL_R / (GEAR_RATIO * ETA)
    return T_motor
```

#### 2.2.2 动力学模型

##### 运动微分方程
```python
def dynamics(self, t, y):
    pos, vel = y
    # 驱动力
    F_drive = self.N_MOTORS * self.MOTOR_T * self.GEAR_RATIO * self.ETA / r
    # 重力分量
    F_grav = m * g * np.sin(self.beta)
    # 摩擦力（上限）
    F_fric_max = self.MU * self.grip_force * self.N_MOTORS
    # 净力
    F_net = min(F_drive, F_fric_max) - F_grav
    # 非线性阻尼：使速度收敛到稳态
    v_ss = max(self._v_ss, 0.01)
    B = F_net / v_ss if F_net > 0 else 50.0
    F_damp = -B * vel
    # 加速度
    acc = (F_net + F_damp) / m
    return [vel, acc]
```
